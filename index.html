function submitToWA() { const subProduct = document.getElementById('sub-product-select').value; const buyerName = document.getElementById('buyer-name').value; const num = document.getElementById('buyer-number').value; const email = document.getElementById('buyer-email').value; const addr = document.getElementById('buyer-address').value; // Validasi sederhana if(!subProduct || !buyerName || !num) { alert("Isi data wajib: Nama, WhatsApp, dan Pilih Layanan"); return; } const cleanNum = num.replace(/\D/g, ''); // Format pesan SIMPLE const msg = `ORDER MFH STORE\n` + `Produk: ${currentProduct}\n` + `Layanan: ${subProduct}\n` + `Nama: ${buyerName}\n` + `WA: ${cleanNum}\n` + `Email: ${email || '-'}\n` + `Catatan: ${addr || '-'}\n\n` + `Siap proses 20-30 menit!`; const whatsappUrl = `https://wa.me/6281998593873?text=${encodeURIComponent(msg)}`; window.open(whatsappUrl, '_blank'); // Update UI document.querySelectorAll('.checkout-progress .step').forEach((step, index) => { step.classList.remove('active'); if (index === 2) step.classList.add('active'); }); // Reset form setelah 2 detik setTimeout(() => { document.getElementById('buyer-name').value = ''; document.getElementById('buyer-number').value = ''; document.getElementById('buyer-email').value = ''; document.getElementById('buyer-address').value = ''; }, 2000); }
